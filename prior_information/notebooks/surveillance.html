def get_support(sparse_coeffs, cc=0.2):
    SpaCoe0 = sparse_coeffs.T
    Supp0 = sp.ones(SpaCoe0.shape)
    indx00 = abs(SpaCoe0) <= cc
    indx10 = abs(SpaCoe0) > cc
    Supp0[indx00] = 0
    Supp_true0 = sp.zeros((Supp0.shape[0],1)) # true support
    for i in range(Supp0.shape[0]):
        if sp.linalg.norm(Supp0[i,:]) > 0:
            Supp_true0[i] = 1
    return Supp0
